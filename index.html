folders.next() : DriveApp.createFolder(folderName); var fileUrls = {}; if (data.files) { data.files.forEach(function(file) { var blob = Utilities.newBlob(Utilities.base64Decode(file.data), "image/jpeg", file.fileName); var driveFile = folder.createFile(blob); fileUrls[file.name] = driveFile.getUrl(); }); } // Guardar en la Hoja con estructura fija de 8 hijos para no romper el orden sheet.appendRow([ new Date(), // A: Marca Temporal data.nombre_el, data.nacimiento_el, // B, C data.nombre_ella, data.nacimiento_ella, // D, E data.comunidad, data.aniversario, // F, G data.whatsapp_el, data.whatsapp_ella, // H, I fileUrls.file_el || "", fileUrls.file_ella || "", fileUrls.file_pareja || "", // J, K, L // Bloques de Hijos (3 columnas por hijo x 8 hijos = 24 columnas) data.hijo1_nombre || "", data.hijo1_nacimiento || "", fileUrls.file_hijo1 || "", data.hijo2_nombre || "", data.hijo2_nacimiento || "", fileUrls.file_hijo2 || "", data.hijo3_nombre || "", data.hijo3_nacimiento || "", fileUrls.file_hijo3 || "", data.hijo4_nombre || "", data.hijo4_nacimiento || "", fileUrls.file_hijo4 || "", data.hijo5_nombre || "", data.hijo5_nacimiento || "", fileUrls.file_hijo5 || "", data.hijo6_nombre || "", data.hijo6_nacimiento || "", fileUrls.file_hijo6 || "", data.hijo7_nombre || "", data.hijo7_nacimiento || "", fileUrls.file_hijo7 || "", data.hijo8_nombre || "", data.hijo8_nacimiento || "", fileUrls.file_hijo8 || "", fileUrls.file_familia || "", data.lema // AK, AL: Foto Familiar y Lema siempre al final ]); return ContentService.createTextOutput(JSON.stringify({"result": "success"})).setMimeType(ContentService.MimeType.JSON); } catch (f) { return ContentService.createTextOutput(JSON.stringify({"result": "error", "error": f.toString()})).setMimeType(ContentService.MimeType.JSON); } }
