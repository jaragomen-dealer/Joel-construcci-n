function doPost(e) { try { var data = JSON.parse(e.postData.contents); var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); // 1. Crear o buscar carpeta para las fotos var folderName = "Fotos Pastoral 2026"; var folders = DriveApp.getFoldersByName(folderName); var folder = folders.hasNext() ? folders.next() : DriveApp.createFolder(folderName); // 2. Procesar Archivos y obtener sus links var fileUrls = {}; if (data.files) { data.files.forEach(function(file) { var blob = Utilities.newBlob(Utilities.base64Decode(file.data), "image/jpeg", file.fileName); var driveFile = folder.createFile(blob); fileUrls[file.name] = driveFile.getUrl(); }); } // 3. Guardar en la Hoja (Ordenado por columnas) sheet.appendRow([ new Date(), data.nombre_el, data.nombre_ella, data.comunidad, data.aniversario, data.whatsapp, fileUrls.file_el || "", fileUrls.file_ella || "", fileUrls.file_pareja || "", data.hijo1_nombre, data.hijo1_cumple, fileUrls.file_hijo1 || "", data.hijo2_nombre, data.hijo2_cumple, fileUrls.file_hijo2 || "", fileUrls.file_familia || "", data.lema ]); return ContentService.createTextOutput(JSON.stringify({"result": "success"})).setMimeType(ContentService.MimeType.JSON); } catch (f) { return ContentService.createTextOutput(JSON.stringify({"result": "error", "error": f.toString()})).setMimeType(ContentService.MimeType.JSON); } }
